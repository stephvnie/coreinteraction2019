window.addEventListener('load', function(e) { // attaches an event handler to the window object, in this case, its to load the following functions.
var appleGrid = document.getElementById('apple-grid');// to access an HTML element using the id of the element.
var appleList = document.getElementById('apple-list');
var appleThumbs = document.getElementById('apple-thumbs');
var gridApples = document.querySelectorAll('.apple');// returns all elements in the document that matches a specified CSS selector(s), as a static NodeList object. The NodeList object represents a collection of nodes. The nodes can be accessed by index numbers. The index starts at 0.

gridApples.forEach((gridApple, i) => {
gridApple.addEventListener('click', pickApple);//same thing becasue its also addEventListener, but this time it actives it when on click.
});

function pickApple(e) {
var apple = e.target;//the target event property returns the element that triggered the event.
var appleName = apple.dataset.apple.toString();//representing apple and appleName as a string

var appleListItem = document.createElement('li');//creates appleListItem with name "list"

var ol = appleList.querySelector('ol');//returns appleList that matches a specified CSS selector, "ol"
ol.appendChild(appleListItem);//adds appleListItem to ol.appendChild
appleListItem.innerText = appleName;
var appleThumb = apple.cloneNode(true);//clones appleThumb.

appleThumbs.appendChild(appleThumb);//adds appleThumbs to appendChild

}
});
